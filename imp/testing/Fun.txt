let Fun = fix \fun :
  { compose : (b -> c) -> (a -> b) -> a -> c
  , curry   : ((a, b) -> c) -> a -> b -> c
  , uncurry : (a -> b -> c) -> (a, b) -> c
  , swap    : (a -> b -> c) -> b -> a -> c
  }.

{ compose = \g : b -> c. \f : a -> b. \x : a. g (f x)

, curry = \f : (a, b) -> c. \a : a. \b : b. f (a, b)

, uncurry = \f : a -> b -> c. \p : (a, b). f (fst p) (snd p)

, swap = \f : a -> b -> c. \b : b. \a : a. f a b
}
